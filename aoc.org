# Lowercase item types a through z have priorities 1 through 26.
# Uppercase item types A through Z have priorities 27 through 52.
def priority(letter):
    val = ord(letter)
    if letter.islower():
        return val - ord('a') + 1
    return val - ord('A') + 27
