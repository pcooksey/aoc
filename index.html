# Lowercase item types a through z have priorities 1 through 26.
# Uppercase item types A through Z have priorities 27 through 52.
def priority(letter):
    val = ord(letter)
    if letter.islower():
        return val - ord('a') + 1
    return val - ord('A') + 27

def move_me(direction, distance, rope, positions):
    move = 1 if direction in ('R', 'U') else -1
    index = 0 if direction in ('R', 'L') else 1
    for _ in range(0, distance):
        rope[0][index] += move
        for knot in range(1, len(rope)):
            head = rope[knot - 1]
            tail = rope[knot]
            if math.hypot(head[0] - tail[0], head[1] - tail[1]) >= 2:
                if head[0] == tail[0] or head[1] == tail[1]: # straight move
                    rope[knot][0] += max(-1, min(1, head[0] - tail[0]))
                    rope[knot][1] += max(-1, min(1, head[1] - tail[1]))
                else: # diagonal move
                    rope[knot][0] += 1 if head[0] - tail[0] > 0 else -1
                    rope[knot][1] += 1 if head[1] - tail[1] > 0 else -1
        positions.add('.'.join(map(str, rope[-1])))

class Program:
    def __init__(self, checks):
        self.x = 1
        self.current_cycle = 1
        self.checks = checks
        self.signal_strength = 0
        self.image = ''
        self.image_cycle = [40, 80, 120, 160, 200, 240]
        self.image_cycle_current = 0

    def cycle(self):
        if self.current_cycle in self.checks:
            self.signal_strength += self.current_cycle * self.x
        if self.x <= self.current_cycle - self.image_cycle_current <= self.x + 2:
            self.image += "#"
        else:
            self.image += '.'
        if self.current_cycle in self.image_cycle:
            self.image += '\n'
            self.image_cycle_current = self.current_cycle
        self.current_cycle += 1

    def process_data(self):
        for line in data.split('\n'):
            cmd = line.split()
            if len(cmd) == 1:  # noop
                self.cycle()
            else:
                self.cycle()
                self.cycle()
                self.x += int(cmd[1])

runner = Program([20, 60, 100, 140, 180, 220])
runner.process_data()
print(runner.signal_strength)

import copy
class Monkey:

    def __init__(self):
        self.items = []
        self._if_true = None
        self._if_false = None
        self.inspected = 0

    def inspect(self):
        if len(self.items):
            self.inspected += 1
            item = self.items.pop(0)
            item = self.operation(item)
            return item
        return None

    def throw(self, item):
        return self._if_true if self.test(item) else self._if_false

class Game:

    def __init__(self, relief):
        self.monkeys = []
        self.relief = relief
        self.gcd = 1

    def add_monkey(self, monkey):
        monkey.relief = self.relief
        self.monkeys.append(monkey)

    def run(self, rounds):
        for r in range(rounds):
            for i, monkey in enumerate(self.monkeys):
                 for _ in range(len(monkey.items)):
                     item = monkey.inspect()
                     if item is not None:
                        if self.relief:
                            item = item // 3
                        else:
                            item = item % self.gcd
                        index = monkey.throw(item)
                        self.monkeys[index].items.append(item)

def create_game(relief):

        if "Monkey" in line:

        elif "Starting" in line:

            items = items.split(',')
            monkey.items.extend([int(i) for i in items])
        elif "Operation" in line:
            _, equation = line.split('=')
            def operation(e):
                return lambda old: eval(f'{e}')
            setattr(monkey, 'operation', operation(equation))
        elif "Test:" in line:
            equation = line.split()
            def test(e):

        elif "If true:" in line:
            monkey._if_true = int(line.split()[-1])
        elif "If false:" in line:
            monkey._if_false = int(line.split()[-1])
        elif "" == line:
            game.add_monkey(monkey)
    game.add_monkey(monkey)
    return game

def generate_moves(p):
    moves = [[p[0] + 1, p[1]],
             [p[0]    , p[1] + 1],
             [p[0] - 1, p[1]],
             [p[0]    , p[1] - 1]]
    return list(filter(lambda pos: 0 <= pos[0] < max_x and 0 <= pos[1] < max_y, moves))

def loc(pos):
    return pos[0] * max_y + pos[1]

def get_path(start, end):

    def h(p):
        return math.hypot(p[0] - end[0], p[1] - end[1]) + \
            abs(grid[end[0]][end[1]] - grid[p[0]][p[1]])

    def d(c, n):  # current, neighbor
        jump = grid[n[0]][n[1]] - grid[c[0]][c[1]]
        if jump > 1:
            return math.inf
        return 1 + jump

    def reconstruct_path(came_from, current):
        total_path = [current]
        while loc(current) in came_from:
            current = came_from[loc(current)]
            total_path.append(current)
        return total_path

    open_set = [start]
    came_from = dict()

    g_score = dict() #:= map with default value of Infinity
    g_score[loc(start)] = 0

    f_score = dict() # := map with default value of Infinity
    f_score[loc(start)] = h(start)

    curr = start
    search = [start]
    while len(open_set):
        open_set.sort(key=lambda p: f_score[loc(p)])
        current = open_set.pop(0)
        c_hash = loc(current)
        if current == end:
            return reconstruct_path(came_from, current)

        for neighbor in generate_moves(current):
            tentative_g_score = g_score[c_hash] + d(current, neighbor)
            n_hash = loc(neighbor)
            if tentative_g_score < g_score.get(n_hash, math.inf):
                came_from[n_hash] = current
                g_score[n_hash] = tentative_g_score
                f_score[n_hash] = tentative_g_score + h(neighbor)
                if neighbor not in open_set:
                    open_set.append(neighbor)
    # Failure case
    return None

def generate_packet(packet, index=0):
    top = index == 0
    array = []
    index += 1 if packet[index] == '[' else 0
    while index < len(packet):
        if packet[index] == '[':
            index, sub_array = generate_packet(packet, index)
            array.append(sub_array)
        elif packet[index] == ']':
            if top:
                return array
            else:
                return index, array
        elif packet[index] in (',', ' '):
            pass
        else:
            comma = packet[index:].find(',')
            bracket = packet[index:].find(']')
            if comma > 0 and comma < bracket:
                array.append(int(packet[index:index + comma]))
                index += comma - 1
            elif bracket > 0:
                array.append(int(packet[index:index + bracket]))
                index += bracket - 1
            else:
                array.append(int(packet[index]))
        index += 1
    return array

def drop_sand(grid, sand, min_x, max_x, min_y, max_y):
    x, y = sand
    taken = lambda grid, x, y: is_taken(grid, x, y, min_x, max_x, min_y, max_y)
    while True:
        if not taken(grid, x, y + 1):
            y += 1
        elif not taken(grid, x - 1, y + 1):
            y += 1
            x -= 1
        elif not taken(grid, x + 1, y + 1):
            y += 1
            x += 1
        else:
            grid[x][y] = "o"
            if [x, y] == sand:
                return False
            return True
        if x < min_x or x >= max_x or y < min_y or y >= max_y:
            return False

               r' closest beacon is at x=([-]?[0-9]+), y=([-]?[0-9]+)')

def manhattan_distance(point1, point2):
    return sum([abs(x1 - x2) for x1, x2 in zip(point1, point2)])

def merge(ranges):
    stack = []
    stack.append(ranges[0])
    for i in ranges[1:]:
        if stack[-1][0] <= i[0] <= stack[-1][-1]:
            stack[-1][-1] = max(stack[-1][-1], i[-1])
        else:
            stack.append(i)
    return stack

def find_line(sensors, y):
    ranges = []
    for s in sensors:
        y_dist = abs(s[1] - y)
        x_dist = s[2] - y_dist
        if x_dist > 0:
            ranges.append([s[0] - x_dist, s[0] + x_dist])
    ranges = sorted(ranges)
    ranges = merge(ranges)
    return ranges
