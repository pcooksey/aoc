# Lowercase item types a through z have priorities 1 through 26.
# Uppercase item types A through Z have priorities 27 through 52.
def priority(letter):
    val = ord(letter)
    if letter.islower():
        return val - ord('a') + 1
    return val - ord('A') + 27
score = 0
group_score = 0
groups = []
for line in data.split('\n'):
    # Part 1
    half = len(line) / 2
    first = set(line[:half])
    second = set(line[half:])
    score += priority(first.intersection(second).pop())
    # Part 2
    groups.append(set(line))
ijk = iter(groups)
for i, j, k in zip(ijk, ijk, ijk):
    group_score += priority(i.intersection(j).intersection(k).pop())
print(score)
print(group_score)

def move_me(direction, distance, rope, positions):
    move = 1 if direction in ('R', 'U') else -1
    index = 0 if direction in ('R', 'L') else 1
    for _ in range(0, distance):
        rope[0][index] += move
        for knot in range(1, len(rope)):
            head = rope[knot - 1]
            tail = rope[knot]
            if math.hypot(head[0] - tail[0], head[1] - tail[1]) >= 2:
                if head[0] == tail[0] or head[1] == tail[1]: # straight move
                    rope[knot][0] += max(-1, min(1, head[0] - tail[0]))
                    rope[knot][1] += max(-1, min(1, head[1] - tail[1]))
                else: # diagonal move
                    rope[knot][0] += 1 if head[0] - tail[0] > 0 else -1
                    rope[knot][1] += 1 if head[1] - tail[1] > 0 else -1
        positions.add('.'.join(map(str, rope[-1])))

class Program:
    def __init__(self, checks):
        self.x = 1
        self.current_cycle = 1
        self.checks = checks
        self.signal_strength = 0
        self.image = ''
        self.image_cycle = [40, 80, 120, 160, 200, 240]
        self.image_cycle_current = 0

    def cycle(self):
        if self.current_cycle in self.checks:
            self.signal_strength += self.current_cycle * self.x
        if self.x <= self.current_cycle - self.image_cycle_current <= self.x + 2:
            self.image += "#"
        else:
            self.image += '.'
        if self.current_cycle in self.image_cycle:
            self.image += '\n'
            self.image_cycle_current = self.current_cycle
        self.current_cycle += 1

    def process_data(self):
        for line in data.split('\n'):
            cmd = line.split()
            if len(cmd) == 1:  # noop
                self.cycle()
            else:
                self.cycle()
                self.cycle()
                self.x += int(cmd[1])

runner = Program([20, 60, 100, 140, 180, 220])
runner.process_data()
print(runner.signal_strength)
print(runner.image)

import copy
class Monkey:

    def __init__(self):
        self.items = []
        self._if_true = None
        self._if_false = None
        self.inspected = 0

    def inspect(self):
        if len(self.items):
            self.inspected += 1
            item = self.items.pop(0)
            item = self.operation(item)
            return item
        return None

    def throw(self, item):
        return self._if_true if self.test(item) else self._if_false

class Game:

    def __init__(self, relief):
        self.monkeys = []
        self.relief = relief
        self.gcd = 1

    def add_monkey(self, monkey):
        monkey.relief = self.relief
        self.monkeys.append(monkey)

    def run(self, rounds):
        for r in range(rounds):
            for i, monkey in enumerate(self.monkeys):
                 for _ in range(len(monkey.items)):
                     item = monkey.inspect()
                     if item is not None:
                        if self.relief:
                            item = item // 3
                        else:
                            item = item % self.gcd
                        index = monkey.throw(item)
                        self.monkeys[index].items.append(item)

def create_game(relief):
    game = Game(relief)
    monkey = None
    for line in iter(data.split('\n')):
        if "Monkey" in line:
            monkey = Monkey()
        elif "Starting" in line:
            _, items = line.split(':')
            items = items.split(',')
            monkey.items.extend([int(i) for i in items])
        elif "Operation" in line:
            _, equation = line.split('=')
            def operation(e):
                return lambda old: eval(f'{e}')
            setattr(monkey, 'operation', operation(equation))
        elif "Test:" in line:
            equation = line.split()
            def test(e):
                return lambda v: v % int(e) == 0
            setattr(monkey, 'test', test(equation[3]))
            game.gcd *= int(equation[3])
        elif "If true:" in line:
            monkey._if_true = int(line.split()[-1])
        elif "If false:" in line:
            monkey._if_false = int(line.split()[-1])
        elif "" == line:
            game.add_monkey(monkey)
    game.add_monkey(monkey)
    return game
