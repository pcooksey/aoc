# Lowercase item types a through z have priorities 1 through 26.
# Uppercase item types A through Z have priorities 27 through 52.
def priority(letter):
    val = ord(letter)
    if letter.islower():
        return val - ord('a') + 1
    return val - ord('A') + 27
score = 0
group_score = 0
groups = []
for line in data.split('\n'):
    # Part 1
    half = len(line) / 2
    first = set(line[:half])
    second = set(line[half:])
    score += priority(first.intersection(second).pop())
    # Part 2
    groups.append(set(line))
ijk = iter(groups)
for i, j, k in zip(ijk, ijk, ijk):
    group_score += priority(i.intersection(j).intersection(k).pop())
print(score)
print(group_score)

def get_range(task):
    a, b = [int(i) for i in task.split('-')]
    return set(range(a, b + 1))

subsets = 0
overlap = 0
for line in data.split('\n'):
    a, b = line.split(',')
    a = get_range(a)
    b = get_range(b)
    subsets += a.issubset(b) or b.issubset(a)
    overlap += len(a.intersection(b)) >= 1
print(subsets)
print(overlap)

system = {'/': {'_files': []}}
path = []
sizes = {'/': 0}

def get_current_dir():
    result = system
    for p in path:
        result = result[p]
    return result

def add_size(file_size):
    i = path[0]
    sizes[i] += file_size
    for p in path[1:]:
        i += p + '/'
        if i not in sizes:
            sizes[i] = file_size
        else:
            sizes[i] += file_size

for line in data.split('\n'):
    line = line.split()
    if line[0] == '$':
        cmd = line[1]
        if cmd == 'cd':
            if line[2] == '..':
                path.pop()
            else:
                path.append(line[2])
        elif cmd == 'ls':
            continue
        else:
            print("Missing ", cmd)
    elif line[0] == 'dir':
        current = get_current_dir()
        if line[1] not in current:
            current[line[1]] = {'_files': []}
    else:
        # We are adding files now
        current = get_current_dir()
        current['_files'].append([int(line[0]), line[1]])
        add_size(int(line[0]))

def puzzle1():
    result = 0
    for key, size in sizes.items():
        if size <= 100000:
            result += size
    return result
print(puzzle1())

def puzzle2():
    max_file_system = 70000000
    needed_space = 30000000
    left_overspace = max_file_system - sizes['/']
    space_to_delete = needed_space - left_overspace
    return min(sizes.values(),
               key=lambda x: max_file_system if x < space_to_delete else \
               abs(x - space_to_delete))
print(puzzle2())
