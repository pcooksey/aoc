# Lowercase item types a through z have priorities 1 through 26.
# Uppercase item types A through Z have priorities 27 through 52.
def priority(letter):
    val = ord(letter)
    if letter.islower():
        return val - ord('a') + 1
    return val - ord('A') + 27
score = 0
group_score = 0
groups = []
for line in data.split('\n'):
    # Part 1
    half = len(line) / 2
    first = set(line[:half])
    second = set(line[half:])
    score += priority(first.intersection(second).pop())
    # Part 2
    groups.append(set(line))
ijk = iter(groups)
for i, j, k in zip(ijk, ijk, ijk):
    group_score += priority(i.intersection(j).intersection(k).pop())
print(score)
print(group_score)

def get_range(task):
    a, b = [int(i) for i in task.split('-')]
    return set(range(a, b + 1))

system = {'/': {'_files': []}}

sizes = {'/': 0}

def get_current_dir():
    result = system
    for p in path:
        result = result[p]
    return result

def add_size(file_size):
    i = path[0]
    sizes[i] += file_size
    for p in path[1:]:
        i += p + '/'
        if i not in sizes:
            sizes[i] = file_size
        else:
            sizes[i] += file_size

for line in data.split('\n'):
    line = line.split()
    if line[0] == '$':
        cmd = line[1]
        if cmd == 'cd':
            if line[2] == '..':
                path.pop()
            else:
                path.append(line[2])
        elif cmd == 'ls':
            continue
        else:
            print("Missing ", cmd)
    elif line[0] == 'dir':
        current = get_current_dir()
        if line[1] not in current:
            current[line[1]] = {'_files': []}
    else:
        # We are adding files now
        current = get_current_dir()
        current['_files'].append([int(line[0]), line[1]])
        add_size(int(line[0]))

def puzzle1():
    result = 0
    for key, size in sizes.items():
        if size <= 100000:
            result += size
    return result
print(puzzle1())

def puzzle2():
    max_file_system = 70000000
    needed_space = 30000000
    left_overspace = max_file_system - sizes['/']
    space_to_delete = needed_space - left_overspace
    return min(sizes.values(),
               key=lambda x: max_file_system if x < space_to_delete else \
               abs(x - space_to_delete))
print(puzzle2())

def get_score(height, arr):
    score = 0
    for a in arr:
        score += 1
        if a >= height:
            return max(1, score)
    return max(1, score)

def move_me(direction, distance, rope, positions):
    move = 1 if direction in ('R', 'U') else -1
    index = 0 if direction in ('R', 'L') else 1
    for _ in range(0, distance):
        rope[0][index] += move
        for knot in range(1, len(rope)):
            head = rope[knot - 1]
            tail = rope[knot]
            if math.hypot(head[0] - tail[0], head[1] - tail[1]) >= 2:
                if head[0] == tail[0] or head[1] == tail[1]: # straight move
                    rope[knot][0] += max(-1, min(1, head[0] - tail[0]))
                    rope[knot][1] += max(-1, min(1, head[1] - tail[1]))
                else: # diagonal move
                    rope[knot][0] += 1 if head[0] - tail[0] > 0 else -1
                    rope[knot][1] += 1 if head[1] - tail[1] > 0 else -1
        positions.add('.'.join(map(str, rope[-1])))

for line in data.split('\n'):
    direction, distance = line.split()
    move_me(direction, int(distance), rope_2, positions_2)
    move_me(direction, int(distance), rope_10, positions_10)
